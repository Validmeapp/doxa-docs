## Step 5: Decrypt Data

The payload is encrypted with your **public key**. You must decrypt it with your **private key**.

### Decrypt with JavaScript/Node.js

```javascript
const crypto = require('crypto');
const fs = require('fs');

function decrypt(encryptedBuffer, privateKeyPath) {
  // Read private key
  const privateKey = fs.readFileSync(privateKeyPath, 'utf8');
  
  // Buffer comes in 256-byte blocks (RSA 2048)
  const blockSize = 256;
  const blocks = [];
  
  for (let i = 0; i < encryptedBuffer.length; i += blockSize) {
    blocks.push(encryptedBuffer.subarray(i, i + blockSize));
  }
  
  // Decrypt each block
  const decryptedBlocks = blocks.map(block => {
    return crypto.privateDecrypt(
      {
        key: privateKey,
        padding: crypto.constants.RSA_PKCS1_OAEP_PADDING,
      },
      block
    );
  });
  
  return Buffer.concat(decryptedBlocks);
}
```

### Decrypt with Python

```python
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.backends import default_backend
import base64
import json

def decrypt(encrypted_payload, private_key_path):
    # Load private key
    with open(private_key_path, 'rb') as f:
        private_key = serialization.load_pem_private_key(
            f.read(),
            password=None,
            backend=default_backend()
        )
    
    # Convert payload from base64 to bytes
    encrypted_data = base64.b64decode(encrypted_payload)
    
    # Decrypt (256 bytes per RSA 2048 block)
    block_size = 256
    decrypted_blocks = []
    
    for i in range(0, len(encrypted_data), block_size):
        block = encrypted_data[i:i + block_size]
        decrypted_block = private_key.decrypt(
            block,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )
        decrypted_blocks.append(decrypted_block)
    
    # Combine blocks
    decrypted_data = b''.join(decrypted_blocks)
    return json.loads(decrypted_data.decode('utf-8'))
```

## Step 7: Grant Access

Based on the result, grant or deny access.

```javascript
// Example: Create user after successful validation
async function processValidation(decryptedData) {
  // 1. Validate confidence
  if (decryptedData.confidence < 0.95) {
    throw new Error('Low confidence');
  }
  
  // 2. Check if user already exists
  const existingUser = await User.findOne({
    documentNumber: decryptedData.documentNumber
  });
  
  if (existingUser) {
    return existingUser;
  }
  
  // 3. Create user
  const user = await User.create({
    firstName: decryptedData.firstName,
    lastName: decryptedData.lastName,
    documentNumber: decryptedData.documentNumber,
    birthDate: decryptedData.birthDate,
    country: decryptedData.country,
    verified: true,
    verifiedAt: new Date(),
    validationId: requestId
  });
  
  // 4. Generate access token
  const token = jwt.sign(
    { userId: user.id },
    process.env.JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  return { user, token };
}
```
